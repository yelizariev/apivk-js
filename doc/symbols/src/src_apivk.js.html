 16 /**
 17  * Create new object to call Vkontakte API throurh it.
 18  *
 19  * @class
 20  * @param {String} api_secret Application sercet (can be
 21  * changed on application edit page).
 22  * @param {Function} success Function to perform after API
 23  * initialised.
 24  * @param {Function} error Function to perform if API
 25  * initialisation fails for msome reason.
 26  * @param {Boolean} test_mode If set to true then all
 27  * requests will be executed in test mode - additional parameter
 28  * test_mode=1 will be automatically added to every API request, so
 29  * you needn't do it manually.
 30  */
 31 function APIVK(api_secret, success, error, test_mode) {
 32 	/****************** Private Constants and variables *******************/
 33 	/**
 34 	 * @constant
 35 	 * @private
 36 	 */
 37 	const vk_js_xd_connection = 'http://vk.com/js/xd_connection.js';
 38 	/**
 39 	 * @constant
 40 	 * @private
 41 	 */
 42 	const vk_js_lib_md5 = 'http://vk.com/js/lib/md5.js';
 43 	/**
 44 	 * @constant
 45 	 * @private
 46 	 */
 47 	const api_version = '2.0';
 48 	/**
 49 	 * Queue of callback functions used by addCallback() to perform correct
 50 	 * processing of multiple events assigned in different places
 51 	 * @private
 52 	 */
 53 	var callbacks = {};
 54 	/**
 55 	 * Queue of api request used when handle error 6 "too many
 56 
 57 	 * requests per second"
 58 	 * @private
 59 	 */
 60 	var apiQueue = [];
 61 	/**
 62 	 * timer delay, when got error 6
 63 	 * @private
 64 	 */
 65 	var timer;
 66 	/**
 67 	 * @private
 68 	 */
 69 	var timerIntervar = 400;
 70 	/**
 71 	 * set of api response handlers unique for each api request
 72 	 * @private
 73 	 */
 74 	apivkHandlers = {};
 75 	/**
 76 	 * num of api request used for make unique name callback function
 77 	 * in api request
 78 	 * @private
 79 	 */
 80 	var apiHandlerNum = 0;
 81 	/**
 82 	 * Hack to use 'this' reference in callback functions in private methods
 83 	 * @private
 84 	 */
 85 	var this_proxy = this;
 86 	/***************** /Private Constants and variables *******************/
 87 
 88 	/************************** Public Constants **************************/
 89 	/**
 90 	 * There is set of settings constants to modify application permissions
 91 	 * using external.showSettingsBox() method/

154 	/**
155 	 * Sort key-value pairs in objects by alphaber in ascending order.
156 	 *
157 	 * @private
158 	 * @param {Object} obj Object which keys to sort.
159 	 * @returns {Object} Object with the same set key-value pairs as 'obj' but
160 	 *                sorted in ascending order.
161 	 */
162 	function _sortByKey(obj) {
163 
164 		//make Array from object keys
165 		var keys = new Array();
166 		for (var k in obj) {
167 			keys.push(k);
168 		}
169 
170 		//sort Array
171 		keys.sort();
172 
173 		//form new object with keys sorted alphabetically
174 		var sortedObj = {};
175 		for (var i = 0; i < keys.length; i++) {
176 			sortedObj[keys[i]] = obj[keys[i]];
177 		}
178 
179 		return sortedObj;
180 	}
181 	/**
182 	 * Perform XSS-request (load JavaScript).
183 	 *
184 	 * @private
185 	 * @param String url URL of JavaScript file to load and execute in
186 	 * context of this page.
187 	 */
188 	function _reqScript(url) {
189 		var script = document.createElement('script');
190 		script.type = 'text/javascript';
191 		script.src = url;
192 		document.getElementsByTagName('head')[0].appendChild(script);
193 	}
194 	/**
195 	 * @private
196 	 */
197 	function _onError6(args){
198 		if (timer)
199 			apiQueue.push(args);
200 		else{
201 			timer = setInterval(_onTimer, timerIntervar);
202 			_call(args);
203 		}
204 	}
205 	/**
206 	 * @private
207 	 */
208 	function _onTimer(){
209 		args = apiQueue.shift(); //fifo
210 		_call(args);
211 		if (apiQueue.length == 0){
212 			clearInterval(timer);
213 			timer = null;
214 		}
215 	}
216 	/**
217 	 * @private
218 	 */
219 	function _call(args) {
220 		var method = args[0];
221 		var req = (args.length>1)? args[1]:{};
222 
223 		var hNum = apiHandlerNum++;
224 		apivkHandlers['func' + hNum] = function(json) {
225 			delete apivkHandlers['func'+hNum];
226 			if (json.response)
227 				if (req.success)
228 					req.success(json.response);
229 			if (json.error){
230 				json = json.error;
231 				if (json.error_code == 6)
232 					_onError6(args);
233 				else if (req.error)
234 					req.error(json.error_code, json.error_msg, json.request_params)
235 			}
236 		};
237 
238 
239 		//parameters to send to server
240 		var parameters = (req.params)? req.params : {};
241 
242 		//append parameters with common strings
243 		parameters.api_id = this_proxy.params.api_id;
244 		parameters.v = api_version;
245 		parameters.format = 'json';
246 		parameters.callback = 'apivkHandlers.func'+hNum;
247 
248 		//set test mode
249 		if (test_mode) parameters.test_mode = '1';
250 
251 		//append parameters with method name
252 		parameters.method = method;
253 
254 		//sort parameters
255 		parameters = _sortByKey(parameters);
256 
257 
258 		//calculate API signature and add it to parameters list
259 		var sig = this_proxy.params.viewer_id;
260 		for (var k in parameters) {
261 
262 			//can be array
263 			if (typeof parameters[k] == 'object') {
264 				sig += k + '=' + parameters[k].join(',');
265 
266 				//ordinar parameter
267 			} else {
268 				sig += k + '=' + parameters[k];
269 			}
270 		}
271 		sig += api_secret;
272 		parameters.sig = MD5(sig);
273 
274 		//form URL to call
275 		var url = this_proxy.params.api_url + '?';
276 		for (var k in parameters) {
277 			url += k + '=' + encodeURIComponent(parameters[k]) + '&';
278 		}
279 
280 
281 		//load and execute script
282 		_reqScript(url);
283 	}
284 
285 	/**** Public: ****/
286 	/**
287 	 * Call Vkontakte API method.
288 	 * @param {String} method Method name to execute.
289 	 * @param {Object} [req] see below
290 	 * @param {Object} req.params Parameters of API method in the following
291 	 *                          form:
292 	 *                          {
293 	 *                              name1: 'value 1',
294 	 *                              name2: 'value 2',
295 	 *
296 	 *                              //arrays are also allowed
297 	 *                              name3: [
298 	 *                                  'array_value_1',
299 	 *                                  'array_value_2',
300 	 *                                  'array_value_3'
301 	 *                              ]
302 	 *                          }
303 	 * @param {Function} req.success Callback function to perform when
304 	 * answer from API call comes from server.  success(data), data -
305 	 * content of "response" json node
306 	 * @param {Function} req.error Callback function to perform when
307 	 * answer from API call comes from server.  error(code, msg,
308 	 * reqParams)
309 	 */
310 	this.call = function(method/*, req*/){
311 		if (timer)
312 			apiQueue.push(arguments);
313 		else
314 			_call(arguments);
315 	}

319 	/**
320 	 * @private
321 	 */
322 	var test_mode  = test_mode;
323 	/**
324 	 * @private
325 	 */
326 	var api_secret = api_secret;
327 	/**
328 	 * on success init
329 	 * @private
330 	 */
331 	var onSuccess  = success;
332 	/**
333 	 * on init failure
334 	 * @private
335 	 */
336 	var onError    = error;
337 	//request MD5 function from vkontakte
338 	_reqScript(vk_js_lib_md5);
339 
340 	//first of all we need to load special library from vk.com
341 	_reqScript(vk_js_xd_connection);
342 
343 	/**
344 	 * wait to 'APIVK_ibnitializer' to load
345 	 * @private
346 	 */
347 	var VKLOAD_intervel = setInterval(apivkInit, 100);
348 
349 	/**
350 	 * Perform when 'APIVK_initializer' loaded.
351 	 * @private
352 	 */
353 	function apivkInit() {
354 		if ((typeof VK == 'undefined') || (typeof MD5 == 'undefined'))
355 			return;
356 		clearInterval(VKLOAD_intervel);
357 
358 		//initialize VK object
359 		VK.init(function() {
360 				//Perform on successful
361 
362 				//get VK.params and translate it to APIVK.params
363 				VK.loadParams(document.location.href);
364 				this_proxy.params = VK.params;
365 
366 				//parse json response from the first api call result if
367 				//it is not in XML
368 				if ((typeof this_proxy.params.api_result != 'undefined') &&
369 				    (this_proxy.params.api_result.indexOf('<?xml') != 0)
370 				    ) {
371 					this_proxy.params.api_result = eval('(' + this_proxy.params.api_result + ')');
372 				}
373 
374 				/**
375 				 * map VK.External methods to APIVK.external
376 				 * @memberOf APIVK#
377 				 * @name external
378 				 * @type Object
379 				 */
380 				this_proxy.external = VK.External;
381 
382 				/**
383 				 * Call external method.
384 				 *
385 				 * @methodOf APIVK#
386 				 * @name callMethod
387 				 * @param String method Method name to call. You can
388 				 *                      call methods from external.*
389 				 *                      (VK.External.*) with proper
390 				 *                      parameters.
391 				 */
392 				this_proxy.callMethod = function(method /*, ...*/) {
393 					VK.callMethod.apply(VK, arguments);
394 				}
395 
396 				/**
397 				 * Add callback function.
398 				 *
399 				 * @methodOf APIVK#
400 				 * @name addCallback
401 				 * @param String name Name of event to which callback
402 				 *                    will be added.
403 				 * @param Function callback Callback function to run
404 				 *                          after event 'name' occurs.
405 				 * @return Number ID of assigned callback function.
406 				 */
407 				this_proxy.addCallback = function(name, callback) {
408 					//add callback to the queue
409 					if (typeof callbacks[name] == 'undefined') {
410 						callbacks[name] = new Array();
411 						//add callback using VK
412 						VK.addCallback(name, function() {
413 								/**
414 								 * Function uses callbacks queue to call all
415 								 * functions that were added using addCallback()
416 								 * in order it were added.
417 								 */
418 
419 								//call every callback in the queue
420 								for (var c in callbacks[name]) {
421 
422 									//form callback call with arguments
423 									callbacks[name][c].apply(null, arguments);
424 								}
425 							});
426 					}
427 					callbacks[name].push(callback);
428 					//return so called ID of assigned callback function
429 					//to use it in removeCallback() function
430 					return callbacks[name].length - 1;
431 				};
432 
433 				/**
434 				 * Remove callback function from the queue.
435 				 *
436 				 * @methodOf APIVK#
437 				 * @name removeCallback
438 				 * @param String name Name of event from which callback
439 				 *                    will be removed.
440 				 * @param Number callback_id ID of callback function to
441 				 *                           remove from queue.
442 				 */
443 				this_proxy.removeCallback = function(name, callback_id) {
444 					//do not remove callback using low-level function
445 					//VK.removeCallback(name, callback);
446 
447 					//remove function from the queue
448 					delete callbacks[name].splice(callback_id, 1);
449 				};
450 				//execute user initialisation function
451 				if (onSuccess)
452 					onSuccess();
453 			},
454 			function() {
455 				// Perform on init failure.
456 				if (onError)
457 					onError();
458 			});
459 	}
460 	/*********************** /Initialize object ***************************/
461 }
